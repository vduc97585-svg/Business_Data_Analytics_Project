# Đường dẫn tới thư mục chứa các file CSV
folder_path = "./All_Fuel_Data" 

# Tìm tất cả các file CSV trong thư mục
csv_files = glob.glob(os.path.join(folder_path, "*.csv"))

# Tạo dictionary chứa DataFrame
dataframes = {}

# Đọc từng file và lưu vào dictionary
for file in csv_files:
    # Lấy tên file không có đuôi .csv để đặt tên
    name = os.path.splitext(os.path.basename(file))[0]
    df = pd.read_csv(file)
    dataframes[name] = df

# Duyệt qua từng DataFrame trong dictionary
for name, dataframe in dataframes.items():
    print(f"Đang xử lý DataFrame: {name}")
    for col in ['open', 'high', 'low', 'close']:
        if col in dataframe.columns:  # Kiểm tra nếu cột tồn tại
            mean_value = dataframe[col][dataframe[col] > 0].mean()  # Tính giá trị trung bình của các giá trị > 0
            dataframe[col] = dataframe[col].apply(lambda x: x if x > 0 else mean_value)  # Thay thế giá trị
            print(f"Đã xử lý cột: {col}")
        else:
            print(f"Cột {col} không tồn tại trong DataFrame {name}.")

output_folder = "processed_data_folder"
os.makedirs(output_folder, exist_ok=True)

# Lưu từng DataFrame thành file CSV
for name, dataframe in dataframes.items():
    # Tạo tên file, ví dụ: Brent_Crude_Oil_data_processed.csv
    file_name = f"{name}_processed.csv"
    file_path = os.path.join(output_folder, file_name)

    # Lưu ra file CSV
    dataframe.to_csv(file_path, index=False)

    print(f"Đã lưu: {file_path}")
