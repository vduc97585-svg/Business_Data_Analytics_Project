folder_path = "./processed_data"

    df = pd.read_csv(file)
    dataframes[name] = df

features = ['open', 'high', 'low', 'volume', 'year', 'month', 'day']
target = 'close'

# Dictionary lưu tập train/test cho từng loại nhiên liệu
train_test_splits = {}

for name, df in dataframes.items():
    X = df[features]
    y = df[target]

    split_index = int(len(df) * 0.8)
    X_train, X_test = X[:split_index], X[split_index:]
    y_train, y_test = y[:split_index], y[split_index:]

    train_test_splits[name] = {
        'X_train': X_train,
        'X_test': X_test,
        'y_train': y_train,
        'y_test': y_test
    }

best_models = {}
best_params = {}

param_grid = {
    'n_estimators': [50, 100, 200],
    'max_depth': [5, 10, 15],
    'min_samples_split': [2, 5],
    'max_features': ['sqrt', 'log2', 0.8]
}

for name, splits in train_test_splits.items():
    print(f"\nTối ưu mô hình cho: {name}")
    
    X_train = splits['X_train']
    y_train = splits['y_train']

    rf = RandomForestRegressor(random_state=42)
    
    grid_search = GridSearchCV(
        estimator=rf,
        param_grid=param_grid,
        cv=3,  # 3-fold cross validation
        n_jobs=-1,
        scoring='neg_mean_squared_error'
    )
    
    grid_search.fit(X_train, y_train)
    
    best_models[name] = grid_search.best_estimator_
    best_params[name] = grid_search.best_params_

    print("Tham số tốt nhất:", grid_search.best_params_)

from sklearn.metrics import mean_absolute_error, mean_squared_error, r2_score
import numpy as np
import matplotlib.pyplot as plt

def mean_absolute_percentage_error(y_true, y_pred):
    y_true, y_pred = np.array(y_true), np.array(y_pred)
    non_zero = y_true != 0  # Tránh chia cho 0
    return np.mean(np.abs((y_true[non_zero] - y_pred[non_zero]) / y_true[non_zero])) * 100

for name, splits in train_test_splits.items():
    model = best_models[name]
    y_train = splits['y_train']
    X_test = splits['X_test']
    y_test = splits['y_test']
    
    y_pred = model.predict(X_test)

    print(f"\nĐánh giá mô hình (tối ưu) cho: {name}")
    print("MAE:", round(mean_absolute_error(y_test, y_pred), 6))
    print("MSE:", round(mean_squared_error(y_test, y_pred), 6))
    print("RMSE:", round(np.sqrt(mean_squared_error(y_test, y_pred)), 6))
    print("R²:", round(r2_score(y_test, y_pred), 6))
    mape = mean_absolute_percentage_error(y_test, y_pred)
    print(f"MAPE: {mape:.6f}%")


    plt.figure(figsize=(14, 6))
    plt.plot(y_train.index, y_train.values, label='Giá thực tế (Train)', color='green')

    plt.plot(y_test.index, y_pred, label='Giá dự báo (Random Forest)', color='orange')

    plt.ylabel('Giá đóng cửa')
    plt.legend()
    plt.grid(True)

    plt.savefig(f"forecast_plot_{name}.png", dpi=300, bbox_inches='tight')
    plt.show()
