from sklearn.model_selection import train_test_split

# Định nghĩa các đặc trưng và mục tiêu
features = ['open', 'high', 'low', 'volume', 'year', 'month', 'day']
target = 'close'

# Dictionary để lưu kết quả chia dữ liệu
split_data = {}

# Duyệt qua từng DataFrame trong dictionary
for name, dataframe in df.items():
    print(f"Đang xử lý DataFrame: {name}")
    if all(col in dataframe.columns for col in features + [target]):  # Kiểm tra nếu tất cả cột tồn tại
        X = dataframe[features]
        y = dataframe[target]
        
        # Chia dữ liệu thành tập huấn luyện và tập kiểm tra
        # Tính chỉ số cắt
        split_index = int(len(X) * 0.8)

        X_train, X_test = X[:split_index], X[split_index:]
        y_train, y_test = y[:split_index], y[split_index:]
        
        # Lưu kết quả vào dictionary
        split_data[name] = {
            'X_train': X_train,
            'X_test': X_test,
            'y_train': y_train,
            'y_test': y_test
        }
        print(f"Đã chia dữ liệu cho DataFrame: {name}")
    else:
        print(f"DataFrame {name} không chứa đủ các cột cần thiết.")

from catboost import CatBoostRegressor

# Dictionary để lưu kết quả dự đoán
predictions = {}

# Duyệt qua từng DataFrame trong split_data
for name, data in split_data.items():
    print(f"Đang huấn luyện mô hình cho DataFrame: {name}")
    
    # Lấy dữ liệu huấn luyện và kiểm tra
    X_train, X_test = data['X_train'], data['X_test']
    y_train, y_test = data['y_train'], data['y_test']
    
    # Khởi tạo mô hình
    model = CatBoostRegressor(
        iterations=1000,
        depth=6,
        learning_rate=0.05,  # 0.05
        loss_function='RMSE',
        verbose=0
    )
    
    # Huấn luyện mô hình
    model.fit(X_train, y_train)
 
    
    # Dự đoán trên tập kiểm tra
    y_pred = model.predict(X_test)
    
    # Lưu kết quả dự đoán
    predictions[name] = {
        'y_train': y_train,
        'y_test': y_test,
        'y_pred': y_pred
    }
    print(f"Đã hoàn thành huấn luyện và dự đoán cho DataFrame: {name}")

for name, result in predictions.items():
    print(f"\nĐánh giá mô hình cho DataFrame: {name}")
    
    # Lấy y_test và y_pred
    y_train = result['y_train']
    y_test = result['y_test']
    y_pred = result['y_pred']
    
    # Tính toán các chỉ số đánh giá
    mae = mean_absolute_error(y_test, y_pred)
    mse = mean_squared_error(y_test, y_pred)
    rmse = np.sqrt(mse)
    r2 = r2_score(y_test, y_pred)
    # Tính MAPE (Mean Absolute Percentage Error)
    mape = np.mean(np.abs((y_test - y_pred) / y_test)) * 100

    
    # In kết quả
    print(f"Mean Absolute Error (MAE): {mae:.6f}")
    print(f"Mean Squared Error (MSE): {mse:.6f}")
    print(f"Root Mean Squared Error (RMSE): {rmse:.6f}")
    print(f"Mean Absolute Percentage Error (MAPE): {mape:.6f}%")
    print(f"R-squared (R²): {r2:.6f}")

import pandas as pd
import matplotlib.pyplot as plt

# Kích thước cửa sổ trung bình động


# Duyệt qua từng DataFrame trong dictionary predictions
for name, result in predictions.items():
    print(f"\nTrực quan hóa trung bình động cho DataFrame: {name}")
    
    y_train = result['y_train']
    y_test = result['y_test']
    y_pred = result['y_pred']
    
    # Chuyển về dạng Series nếu cần
    y_train = y_train.reset_index(drop=True)
    y_test = y_test.reset_index(drop=True)

    # Chỉ số bắt đầu của tập test
    split_index = len(y_train)

    # Tính trung bình động
    # Tính trung bình động
    y_train_smoothed = y_train.rolling(window=10).mean()
    y_test_smoothed = y_test.rolling(window=10).mean()
    y_pred_smoothed = pd.Series(y_pred).rolling(window=10).mean()
    
    # Vẽ biểu đồ
    plt.figure(figsize=(12, 4))
    plt.plot(y_train_smoothed, label='Giá thực tế (Train)', color='green')
    plt.plot(range(split_index, split_index + len(y_test_smoothed)), y_test_smoothed, label='Giá thực tế (Test)', color='blue')
    plt.plot(range(split_index, split_index + len(y_pred_smoothed)), y_pred_smoothed, label='Giá dự báo (Catboost)', color='orange')
    
    plt.title(f'Dự báo giá đóng cửa - {name}')
    plt.xlabel('Thời gian')
    plt.ylabel('Giá đóng cửa')
    plt.legend()
    plt.tight_layout()
    plt.grid(True)
    plt.show()
