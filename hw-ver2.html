        # Chia dữ liệu thành đặc trưng (features) và mục tiêu (target)

        # Chia dữ liệu theo thời gian
        X_train = X.iloc[:train_size]
        X_test = X.iloc[train_size:]
        y_train = y.iloc[:train_size]
        y_test = y.iloc[train_size:]

        # Lưu kết quả vào dictionary
        split_data[name] = {
            'X_train': X_train,
            'X_test': X_test,
            'y_train': y_train,
            'y_test': y_test
        }

        # In thông tin số lượng mẫu
        print(f"Tổng số mẫu: {len(dataframe)}")
        print(f"Train set: {len(X_train)} mẫu ({len(X_train)/len(dataframe):.1%})")
        print(f"Test set: {len(X_test)} mẫu ({len(X_test)/len(dataframe):.1%})")
    else:
        print(f"DataFrame {name} không chứa đủ các cột cần thiết.")

def select_holt_winters_model(name, y_train, y_test):
    # Định nghĩa thông số phù hợp theo tên dataset
    model_config = {
        'Brent_Crude_Oil':      {'seasonal': 'mul', 'trend_options': ['add', 'mul'], 'seasonal_periods': 365},
        'Crude_Oil':            {'seasonal': 'add', 'trend_options': ['add'],        'seasonal_periods': 365},
        'Heating_Oil':          {'seasonal': 'mul', 'trend_options': ['add', 'mul'], 'seasonal_periods': 365},
        'Natural_Gas':          {'seasonal': 'mul', 'trend_options': ['add', 'mul'], 'seasonal_periods': 206},
        'RBOB_Gasoline':        {'seasonal': 'mul', 'trend_options': ['add', 'mul'], 'seasonal_periods': 365}
    }

    config = model_config.get(name)
    if config is None:
        return {"Error": f"Không tìm thấy cấu hình cho '{name}'"}

    seasonal_type = config['seasonal']
    trend_options = config['trend_options']
    seasonal_periods = config['seasonal_periods']
    damped_options = [True, False]

    best_score = float('inf')
    best_model = None
    best_forecast = None
    best_params = {}
    fit_count = 0

    adjusted_test = y_test.iloc[-len(y_train):] if len(y_test) > len(y_train) else y_test

    for trend in trend_options:
        for damped in damped_options if trend is not None else [False]:
            if trend == 'mul' and damped:
                continue  # Không hỗ trợ mul + damped
            fit_count += 1

            try:
                model = ExponentialSmoothing(
                    y_train,
                    trend=trend,
                    seasonal=seasonal_type,
                    seasonal_periods=seasonal_periods,
                    damped_trend=damped
                ).fit(optimized=True)

                forecast = model.forecast(len(adjusted_test))
                forecast = np.nan_to_num(forecast, nan=0)
                mae = mean_absolute_error(adjusted_test, forecast)

                if mae < best_score:
                    best_score = mae
                    best_model = model
                    best_forecast = forecast
                    best_params = {
                        'seasonal_periods': seasonal_periods,
                        'trend': trend,
                        'seasonal_type': seasonal_type,
                        'damped': damped,
                        'alpha': model.params.get('smoothing_level', None),
                        'beta': model.params.get('smoothing_slope', None),
                        'gamma': model.params.get('smoothing_seasonal', None)
                    }

            except Exception as e:
                continue  # Bỏ qua nếu lỗi

    if best_model is not None:
        residuals = best_model.resid.dropna()
        adf_pvalue = adfuller(residuals)[1]

        mse = mean_squared_error(y_test, best_forecast)
        test_values = y_test.replace(0, np.nan)
        mape = np.mean(np.abs((test_values - best_forecast) / test_values)) * 100 if test_values.notna().sum() > 0 else np.nan
        if np.isnan(mape):
            mape = best_score

        r2 = r2_score(y_test, best_forecast)

        metrics = {
            'MAE': best_score,
            'RMSE': np.sqrt(mse),
            'MSE': mse,
            'MAPE': mape,
            'R2': r2,
            'AIC': best_model.aic,
            'BIC': best_model.bic,
            'ADF_pvalue': adf_pvalue,
            'Fit_Count': fit_count
        }

        return {
            'Best_Model': best_model,
            'Best_Forecast': best_forecast,
            'Best_Params': best_params,
            'Metrics': metrics,
            'Residuals': residuals
        }
    else:
        return {"Error": f"Không tìm được mô hình tốt cho '{name}'"}

hw_predictions = {}

    y_train, y_test = data['y_train'], data['y_test']

    # Gọi hàm chọn mô hình tốt nhất Holt-Winters

        y_train,

    # Lưu kết quả dự đoán
    hw_predictions[name] = result
    print(f"Đã hoàn thành mô hình Holt-Winters cho DataFrame: {name}")

for name, result in hw_predictions.items():
    if 'Error' in result:

        continue

    data = split_data[name]
    y_train = data['y_train']
    y_test = data['y_test']
    forecast_values = result['Best_Forecast'][:len(y_test)]

    # Trục thời gian hay số
    if use_time_axis and isinstance(y_train.index, pd.DatetimeIndex):
        x_train = y_train.index
        x_test = y_test.index
    else:
        x_train = range(len(y_train))
        x_test = range(len(y_train), len(y_train) + len(y_test))

    # Vẽ biểu đồ với format giống ảnh
    plt.figure(figsize=(14, 6))

    # Vẽ train (giá thực tế tập huấn luyện) - màu xanh lá
    plt.plot(x_train, y_train, label='Giá thực tế (Train)', color='green')

    # Vẽ test (giá thực tế tập kiểm tra) - màu xanh dương
    plt.plot(x_test, y_test, label='Giá thực tế (Test)', color='blue')

    # Vẽ forecast (giá dự báo) - màu cam
    plt.plot(x_test[:len(forecast_values)], forecast_values,
             label='Giá dự báo (Holt-Winters)', color='orange')

    # Tiêu đề và nhãn
    plt.title(f'Dự báo giá đóng cửa - {name}')
    plt.xlabel('Thời gian')
    plt.ylabel('Giá Đóng Cửa')

    # Hiển thị chú thích và lưới
    plt.legend()
    plt.grid(True)
    plt.tight_layout()

    # Định dạng trục thời gian nếu có
